Function converts dash/underscore delimited words into camel casing. The first word within the output should be capitalized only if the original word was capitalized. The next words should be always capitalized. 

    function toCamelCase(str){
    
    let result = str.replace(/-|_/gi,' ').split(' ');
    for (i = 1; i < result.length; i++){
    result[i] = result[i].charAt(0).toUpperCase() + result[i].slice(1);
    }
    return result.join('');
    }
